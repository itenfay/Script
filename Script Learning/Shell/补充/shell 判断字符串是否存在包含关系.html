shell 判断字符串是否存在包含关系

(2013-08-19 23:18:05)
转载▼

#! /bin/bash

var1="hello"
var2="he"

#方法1
if [ ${var1:0:2} = $var2 ]
then
    echo "1:include"
fi

#方法2
echo "$var1" |grep -q "$var2"
if [ $? -eq 0 ]
then
    echo "2:include"
fi

#方法3
echo "$var1" |grep -q "$var2" && echo "include" ||echo "not"

#方法4
[[ "${var1/$var2/}" != "$var2" ]] && echo "include" || echo "not"

expr或awk的index函数
${var#...}                  
${var%...}
${var/.../...}
